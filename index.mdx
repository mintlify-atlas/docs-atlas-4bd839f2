      <h2 className="text-3xl font-bold dark:text-white text-gray-900 mb-4">Quick Start</h2>
      <p className="text-base dark:text-gray-400 text-gray-600">Get EduStack Smart running in minutes</p>
    </div>
    
    <Steps>
      <Step title="Clone the repository">
        Clone the EduStack Smart repository to your local machine:
        
        ```bash
        git clone https://github.com/DiegoMP06/innotech-app.git
        cd innotech-app
        ```
      </Step>
      
      <Step title="Install dependencies">
        Install both PHP and JavaScript dependencies:
        
        <CodeGroup>
        ```bash Composer & NPM
        composer install
        npm install
        ```
        
        ```bash Composer & PNPM
        composer install
        pnpm install
        ```
        </CodeGroup>
      </Step>
      
      <Step title="Configure environment">
        Copy the example environment file and configure your database:
        
        ```bash
        cp .env.example .env
        php artisan key:generate
        ```
        
        <Note>Update the `.env` file with your database credentials and application settings.</Note>
      </Step>
      
      <Step title="Run migrations">
        Set up the database schema and seed initial data:
        
        ```bash
        php artisan migrate --seed
        ```
      </Step>
      
      <Step title="Start the development server">
        Launch the application with all services:
        
        ```bash
        composer run dev
        ```
        
        This command starts the Laravel server, queue worker, and Vite development server concurrently.
      </Step>
    </Steps>
  </div>
